678. Valid Parenthesis String.
Posted on 2024-04-07

Working but not optimal solution:

This is DP. It can be done top down or bottom up.

class Solution:
    def checkValidString(self, s: str) -> bool:
        @cache
        def dp(ind, l):
            if ind == len(s):
                if l:
                    return False
                return True
            if l < 0:
                return False
            if s[ind] == '(':
                return dp(ind+1, l+1)
            if s[ind] == ')':
                return dp(ind+1, l-1)
            return dp(ind+1, l-1) or dp(ind+1, l+1) or dp(ind+1, l)

        return dp(0, 0)

Optimal

class Solution:
    def checkValidString(self, s: str) -> bool:
        open_count = 0
        star_count = 0
        
        for char in s:
            if char == '(':
                open_count += 1
            elif char == ')':
                if open_count == 0:
                    star_count -= 1
                else:
                    open_count -= 1
            elif char == '*':
                star_count += 1
            if star_count == -1:
                return False
        if open_count > star_count:
            return False
        
        open_count = 0
        star_count = 0
        
        for char in reversed(s):
            if char == ')':
                open_count += 1
            elif char == '(':
                if open_count == 0:
                    if star_count == 0:
                        return False
                    star_count -= 1
                else:
                    open_count -= 1
            elif char == '*':
                star_count += 1
        
        return open_count <= star_count
